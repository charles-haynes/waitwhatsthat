{{/* this function takes an image resource and returns a string containing image RelPermalinks and widths suitable for using as a srcSet */}} {{- $img := . -}} {{- $sizes := (slice 320 480 768 1024 1280) -}} {{- $srcSet := slice }} {{- range $sizes -}} {{- if lt . $img.Width -}} {{- $srcSet = $srcSet | append (printf "%s %dw" ($img.Resize (printf "%dx q75 Lanczos" .)).RelPermalink .) -}} {{- end -}} {{- end -}} {{- return (delimit $srcSet ", ") -}}
